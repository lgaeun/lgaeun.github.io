import sys
input = sys.stdin.readline

n = int(input())
mat = []
for i in range(n):
    mat.append(list(map(int, input().split())))

dx = [0,0,-1,1]
dy = [-1,1,0,0]

def bfs(sx, sy, visit, limit):
    visit[sx][sy] = True
    q = []
    q.append([sx,sy])
    while q:
        x, y = q.pop(0)
        for i in range(4):
            nx = x + dx[i]
            ny = y + dy[i]
            if nx<0 or nx>=n or ny<0 or ny>=n or visit[nx][ny]: continue
            if mat[nx][ny] > limit:
                visit[nx][ny] = True
                q.append([nx, ny])

ans=0
rain = 0
while True:
    cnt = 1
    visit = [[0] * n for _ in range(n)]

    for i in range(n):
        for j in range(n):
            if (not visit[i][j]) and mat[i][j] > rain:
                bfs(i, j, visit, rain)
                cnt +=1
    if cnt== 1:
        break
    ans = max(ans, cnt-1)
    rain +=1

print(ans)

n = int(input())
ss = []

alpha = [0] * 26

for _ in range(n):
    ss.append(input())

for s in ss:
    base = pow(10, len(s) - 1)
    for i in range(len(s)):
        alpha[ord(s[i]) - ord('A')] += int(base)
        base /= 10

alpha.sort(reverse=True)

ans = 0
for i in range(9, 0, -1):
    ans += i * alpha[9-i]

print(ans)
